Interface ACL<ChangesType, PublicKey>

An ACL keeps track of a list of user's public keys and produces changes that can be sent to other swarm peers.

interface ACL<ChangesType, PublicKey> {
    add(publicKey: PublicKey): Promise<ChangesType>;
    check(publicKey: PublicKey): Promise<boolean>;
    current(): ChangesType;
    merge(changes: ChangesType): void;
    remove(publicKey: PublicKey): Promise<ChangesType>;
    users(): Promise<PublicKey[]>;
}
